GEOGRAPHY||Is Rome the capital of Italy?
SELECT count(*) From Cities C Join Capitals Cap on C.id == Cap.CityID Join Countries Cn on Cn.id == Cap.CountryID Join CountryContinents CC on CC.CountryID == Cn.ID Join Continents Con on Con.id == CC.ContinentID WHERE C.name like "%Rome%" and Cn.name like "%Italy%"  ;
1
